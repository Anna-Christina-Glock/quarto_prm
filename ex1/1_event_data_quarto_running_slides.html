Convertions

CSV to XES

# Read CSV File
df = pd.read_csv('data/running-example.csv', sep=';')
df.Timestamp = pd.to_datetime(df.Timestamp, format="%d-%m-%Y:%H.%M")

# Convert to pm4py log
log = pm4py.format_dataframe(df, case_id='Case ID',activity_key='Activity',
                             timestamp_key='Timestamp')

# Export as XES File                      
pm4py.write_xes(log, 'data/data_out/running-example.xes')


XES to CSV

log_xes = pm4py.read_xes('data/running-example.xes', return_legacy_log_object=True)

df = pm4py.convert_to_dataframe(log_xes)
df.to_csv('data/data_out/running_example_xes_exported_as_csv.csv')
